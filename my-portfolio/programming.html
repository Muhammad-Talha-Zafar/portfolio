class RoboController(Node):
    def __init__(self):
        super().__init__('driver')
        self.pub = self.create_pub(Twist, 'cmd_vel')
        
    def send_velocity(self, x, z):
        msg = Twist()
        msg.linear.x = x
        msg.angular.z = z
        self.pub.publish(msg)
